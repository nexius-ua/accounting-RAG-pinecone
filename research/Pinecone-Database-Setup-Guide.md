### –ö—Ä–æ–∫ 2.4: –°—Ç–≤–æ—Ä—ñ—Ç—å —Å–∫—Ä–∏–ø—Ç –¥–ª—è chunking —Ç–∞ –∑–∞–≤–∞–Ω—Ç–∞–∂–µ–Ω–Ω—è

```python
"""

–ó–∞–ø—É—Å–∫: python scripts/chunk_and_upload.py
"""

import os
import re
import hashlib

# === –ù–ê–õ–ê–®–¢–£–í–ê–ù–ù–Ø (–ó–ú–Ü–ù–Ü–¢–¨ –ü–Ü–î –°–ï–ë–ï) ===
PINECONE_API_KEY = "pcsk_–≤–∞—à_–∫–ª—é—á_—Ç—É—Ç"  # ‚Üê –í—Å—Ç–∞–≤—Ç–µ –≤–∞—à –∫–ª—é—á

CHUNK_SIZE = 500                        # –¢–æ–∫–µ–Ω—ñ–≤ –Ω–∞ chunk (~2000 —Å–∏–º–≤–æ–ª—ñ–≤)
CHUNK_OVERLAP = 50                      # –ü–µ—Ä–µ–∫—Ä–∏—Ç—Ç—è –º—ñ–∂ chunks

# =========================================

def count_tokens_approx(text: str) -> int:
    """–ü—Ä–∏–±–ª–∏–∑–Ω–∏–π –ø—ñ–¥—Ä–∞—Ö—É–Ω–æ–∫ —Ç–æ–∫–µ–Ω—ñ–≤ (1 —Ç–æ–∫–µ–Ω ‚âà 4 —Å–∏–º–≤–æ–ª–∏ –¥–ª—è —É–∫—Ä–∞—ó–Ω—Å—å–∫–æ—ó)"""
    return len(text) // 4

def chunk_text(text: str, chunk_size: int = 500, overlap: int = 50) -> list[str]:
    """
    –†–æ–∑–±–∏–≤–∞—î —Ç–µ–∫—Å—Ç –Ω–∞ chunks –∑ –ø–µ—Ä–µ–∫—Ä–∏—Ç—Ç—è–º.
    –ù–∞–º–∞–≥–∞—î—Ç—å—Å—è —Ä–æ–∑–±–∏–≤–∞—Ç–∏ –ø–æ –∞–±–∑–∞—Ü–∞—Ö/—Ä–µ—á–µ–Ω–Ω—è—Ö, –∞ –Ω–µ –ø–æ—Å–µ—Ä–µ–¥ —Å–ª–æ–≤–∞.
    """
    # –°–ø–æ—á–∞—Ç–∫—É —Ä–æ–∑–±–∏–≤–∞—î–º–æ –ø–æ –∞–±–∑–∞—Ü–∞—Ö
    paragraphs = re.split(r'\n\s*\n', text)
    
    chunks = []
    current_chunk = ""
    
    for para in paragraphs:
        para = para.strip()
        if not para:
            continue
            
        # –Ø–∫—â–æ –∞–±–∑–∞—Ü —Å–∞–º –ø–æ —Å–æ–±—ñ –∑–∞–Ω–∞–¥—Ç–æ –≤–µ–ª–∏–∫–∏–π ‚Äî —Ä–æ–∑–±–∏–≤–∞—î–º–æ –ø–æ —Ä–µ—á–µ–Ω–Ω—è—Ö
        if count_tokens_approx(para) > chunk_size:
            sentences = re.split(r'(?<=[.!?])\s+', para)
            for sentence in sentences:
                if count_tokens_approx(current_chunk + " " + sentence) <= chunk_size:
                    current_chunk = (current_chunk + " " + sentence).strip()
                else:
                    if current_chunk:
                        chunks.append(current_chunk)
                    current_chunk = sentence
        else:
            # –î–æ–¥–∞—î–º–æ –∞–±–∑–∞—Ü –¥–æ –ø–æ—Ç–æ—á–Ω–æ–≥–æ chunk
            if count_tokens_approx(current_chunk + "\n\n" + para) <= chunk_size:
                current_chunk = (current_chunk + "\n\n" + para).strip()
            else:
                if current_chunk:
                    chunks.append(current_chunk)
                current_chunk = para
    
    # –ù–µ –∑–∞–±—É–≤–∞—î–º–æ –æ—Å—Ç–∞–Ω–Ω—ñ–π chunk
    if current_chunk:
        chunks.append(current_chunk)
    
    # –î–æ–¥–∞—î–º–æ –ø–µ—Ä–µ–∫—Ä–∏—Ç—Ç—è (overlap)
    if overlap > 0 and len(chunks) > 1:
        overlapped_chunks = []
        for i, chunk in enumerate(chunks):
            if i > 0:
                # –î–æ–¥–∞—î–º–æ –∫—ñ–Ω–µ—Ü—å –ø–æ–ø–µ—Ä–µ–¥–Ω—å–æ–≥–æ chunk –Ω–∞ –ø–æ—á–∞—Ç–æ–∫
                prev_words = chunks[i-1].split()[-overlap:]
                chunk = " ".join(prev_words) + " [...] " + chunk
            overlapped_chunks.append(chunk)
        chunks = overlapped_chunks
    
    return chunks

def generate_chunk_id(filename: str, chunk_index: int, text: str) -> str:
    """–ì–µ–Ω–µ—Ä—É—î —É–Ω—ñ–∫–∞–ª—å–Ω–∏–π ID –¥–ª—è chunk"""
    hash_input = f"{filename}_{chunk_index}_{text[:100]}"
    return hashlib.md5(hash_input.encode()).hexdigest()[:16]

def process_file(filepath: Path) -> list[dict]:
    """–û–±—Ä–æ–±–ª—è—î –æ–¥–∏–Ω —Ñ–∞–π–ª: —á–∏—Ç–∞—î, chunking, –ø–æ–≤–µ—Ä—Ç–∞—î –∑–∞–ø–∏—Å–∏ –¥–ª—è Pinecone"""
    print(f"  –û–±—Ä–æ–±–∫–∞: {filepath.name}")
    
    with open(filepath, 'r', encoding='utf-8') as f:
        text = f.read()
    
    chunks = chunk_text(text, CHUNK_SIZE, CHUNK_OVERLAP)
    print(f"    ‚Üí {len(chunks)} chunks")
    
    records = []
```

## –î–æ–¥–∞—Ç–æ–∫ D: –ê–ª—å—Ç–µ—Ä–Ω–∞—Ç–∏–≤–Ω–∏–π —Å–∫—Ä–∏–ø—Ç (–ø—Ä–æ—Å—Ç—ñ—à–∏–π)

–Ø–∫—â–æ –æ—Å–Ω–æ–≤–Ω–∏–π —Å–∫—Ä–∏–ø—Ç –Ω–µ –ø—Ä–∞—Ü—é—î, –æ—Å—å –º—ñ–Ω—ñ–º–∞–ª—å–Ω–∞ –≤–µ—Ä—Å—ñ—è:

```python
"""–ú—ñ–Ω—ñ–º–∞–ª—å–Ω–∏–π —Å–∫—Ä–∏–ø—Ç –¥–ª—è –∑–∞–≤–∞–Ω—Ç–∞–∂–µ–Ω–Ω—è –≤ Pinecone"""

from pinecone import Pinecone
from pathlib import Path

API_KEY = "pcsk_–≤–∞—à_–∫–ª—é—á"
INDEX_NAME = "legal-docs-ua"

pc = Pinecone(api_key=API_KEY)
index = pc.Index(INDEX_NAME)

# –ß–∏—Ç–∞—î–º–æ –≤—Å—ñ .md —Ñ–∞–π–ª–∏
for filepath in Path("source_docs").glob("*.md"):
    text = filepath.read_text(encoding="utf-8")
    
    # –ü—Ä–æ—Å—Ç–∏–π chunking –ø–æ –∞–±–∑–∞—Ü–∞—Ö
    chunks = [p.strip() for p in text.split("\n\n") if p.strip()]
    
    # –ó–∞–≤–∞–Ω—Ç–∞–∂—É—î–º–æ
    for i, chunk in enumerate(chunks):
        if len(chunk) > 100:  # –Ü–≥–Ω–æ—Ä—É—î–º–æ –¥—É–∂–µ –∫–æ—Ä–æ—Ç–∫—ñ
            index.upsert_records(
                namespace="default",
                records=[{
                    "id": f"{filepath.stem}_{i}",
                    "text": chunk,
                    "metadata": {"filename": filepath.name, "chunk": i}
                }]
            )
    
    print(f"‚úì {filepath.name}: {len(chunks)} chunks")

print("–ì–æ—Ç–æ–≤–æ!")
```
